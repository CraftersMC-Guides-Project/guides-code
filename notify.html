Cloudflare Worker Code for KV Storage

This worker handles saving FCM tokens to KV and sending notifications:

// Cloudflare Worker for FCM Token Management

// Define the KV namespace binding
// Make sure to create a KV namespace in your Cloudflare dashboard
// and add it to your wrangler.toml or dashboard bindings

const FCM_TOKENS = "fcm-tokens"; // Your KV namespace

async function handleRequest(request) {
const url = new URL(request.url);
const path = url.pathname;
const method = request.method;

// Handle preflight OPTIONS request for CORS
if (method === 'OPTIONS') {
return handleOptions(request);
}

// Save FCM token to KV
if (method === 'POST' && path === '/save-token') {
try {
const { token, userId } = await request.json();

if (!token) {
return new Response('Token is required', { status: 400 });
}

// Use current timestamp as user ID if not provided
const user = userId || Date.now().toString();

// Save token to KV (store multiple tokens per user)
let userTokens = [];
try {
const existing = await FCM_TOKENS.get(user);
userTokens = existing ? JSON.parse(existing) : [];
} catch (e) {
console.log('No existing tokens for user:', user);
}

// Add new token if not already present
if (!userTokens.includes(token)) {
userTokens.push(token);
}

// Save updated tokens
await FCM_TOKENS.put(user, JSON.stringify(userTokens));

return new Response(JSON.stringify({ success: true, userId: user }), {
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*',
'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
'Access-Control-Allow-Headers': 'Content-Type'
}
});
} catch (error) {
return new Response(JSON.stringify({ error: error.message }), {
status: 500,
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*',
'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
'Access-Control-Allow-Headers': 'Content-Type'
}
});
}
}

// Send notification to all users
if (method === 'POST' && path === '/send-notification') {
try {
const { title, body } = await request.json();

if (!title || !body) {
return new Response('Title and body are required', { status: 400 });
}

// Get all keys from KV
const keys = await FCM_TOKENS.list();

// Send notification to each token
const results = [];
for (const key of keys.keys) {
const tokens = JSON.parse(await FCM_TOKENS.get(key.name));

for (const token of tokens) {
// Send FCM notification using Firebase Admin SDK
// This would typically be done in a Node.js environment
// For demonstration, we'll just log it
console.log(`Sending notification to token: ${token}`);
console.log(`Title: ${title}, Body: ${body}`);
results.push({ token, status: 'simulated' });
}
}

return new Response(JSON.stringify({
success: true,
sent: results.length,
results
}), {
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*',
'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
'Access-Control-Allow-Headers': 'Content-Type'
}
});
} catch (error) {
return new Response(JSON.stringify({ error: error.message }), {
status: 500,
headers: {
'Content-Type': 'application/json',
'Access-Control-Allow-Origin': '*',
'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
'Access-Control-Allow-Headers': 'Content-Type'
}
});
}
}

return new Response('Not found', { status: 404 });
}

// Handle CORS preflight requests
function handleOptions(request) {
return new Response(null, {
headers: {
'Access-Control-Allow-Origin': '*',
'Access-Control-Allow-Methods': 'GET, POST, OPTIONS',
'Access-Control-Allow-Headers': 'Content-Type'
}
});
}

// Add event listener for fetch events
addEventListener('fetch', event => {
event.respondWith(handleRequest(event.request));
});

Make sure to:

  1. Create a KV namespace in your Cloudflare dashboard
  2. Add the binding to your worker (wrangler.toml or dashboard)
  3. Deploy the worker to your Cloudflare account

Implementation Instructions

  4. Update the worker URL in the frontend code below
  5. Deploy both the frontend and worker
